216  @JsonMappingCompleted
217  protected void jsonMappingCompleted(JsonMapper jsonMapper) {
218    messageTags = new HashMap<String, List<MessageTag>>();
219    if (rawMessageTags == null) return;
220    for (String key : getNames(rawMessageTags)) {
221      String messageTagJson = rawMessageTags.getString(key).toString();
222      messageTags.put(key, jsonMapper.toJavaList(messageTagJson, MessageTag.class));
223    }
